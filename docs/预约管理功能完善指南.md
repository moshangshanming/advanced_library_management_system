```python
def export_reservations(self, status_filter: str = "", keyword: str = "", 
                       start_date: str = "", end_date: str = "") -> str:
    """导出预约记录为CSV"""
    conditions: List[str] = []
    params: List[Any] = []
    
    if status_filter.strip():
        conditions.append("r.status = ?")
        params.append(status_filter.strip())
    if keyword.strip():
        conditions.append("(b.title LIKE ? OR u.username LIKE ? OR u.full_name LIKE ?)")
        like = normalize_like(keyword)
        params.extend([like, like, like])
    if start_date:
        conditions.append("r.reserve_date >= ?")
        params.append(start_date)
    if end_date:
        conditions.append("r.reserve_date <= ?")
        params.append(end_date)
    
    where_sql = " WHERE " + " AND ".join(conditions) if conditions else ""
    
    rows = db_manager.fetch_all(f"""
        SELECT 
            r.id,
            b.title AS book_title,
            b.isbn,
            u.full_name AS reader_name,
            u.username,
            u.department,
            r.reserve_date,
            r.expiry_date,
            r.status,
            r.remark
        FROM book_reservations r
        JOIN books b ON r.book_id = b.id
        JOIN users u ON r.reader_id = u.id
        {where_sql}
        ORDER BY r.id
    """, tuple(params))
    
    # 状态映射
    status_map = {
        'pending': '待取书',
        'notified': '已通知',
        'fulfilled': '已取书',
        'cancelled': '已取消',
        'expired': '已过期'
    }
    
    # 转换数据格式
    formatted_rows = []
    for row in rows:
        formatted_rows.append({
            'ID': row['id'],
            '图书名称': row['book_title'],
            'ISBN': row['isbn'] or '',
            '读者姓名': row['reader_name'],
            '用户名': row['username'],
            '院系': row['department'] or '',
            '预约日期': row['reserve_date'],
            '到期日期': row['expiry_date'] or '',
            '状态': status_map.get(row['status'], row['status']),
            '备注': row['remark'] or ''
        })
    
    return self.to_csv(formatted_rows)
```

```javascript
async function submitReservationForm() {
  const bookId = $('reservationBook').value;
  const readerId = $('reservationReader').value;
  const reserveDate = $('reservationDate').value;
  const validDays = $('reservationValidDays').value;
  const expiryDate = $('reservationExpiryDate').value;
  const source = $('reservationSource').value;
  const remark = $('reservationRemark').value;
  const reservationId = $('reservationId').value;
  
  // 表单校验
  if (!bookId) {
    toast('请选择图书', 'warning');
    return;
  }
  if (!readerId) {
    toast('请选择读者', 'warning');
    return;
  }
  if (!reserveDate) {
    toast('请选择预约日期', 'warning');
    return;
  }
  
  // 获取选中的读者信息，检查状态
  const readerSelect = $('reservationReader');
  const selectedOption = readerSelect.options[readerSelect.selectedIndex];
  if (selectedOption.dataset.status && selectedOption.dataset.status !== 'active') {
    toast('该读者账号状态异常，无法预约', 'error');
    return;
  }
  
  try {
    const payload = {
      book_id: parseInt(bookId),
      reader_id: parseInt(readerId),
      reserve_date: reserveDate,
      valid_days: parseInt(validDays),
      expiry_date: expiryDate,
      source: source,
      remark: remark
    };
    
    let result;
    if (reservationId) {
      // 修改预约
      result = await api(`/api/reservations/${reservationId}`, { 
        method: 'PUT', 
        body: JSON.stringify(payload) 
      });
      toast('预约修改成功！', 'success');
    } else {
      // 新增预约
      result = await api('/api/reservations', { 
        method: 'POST', 
        body: JSON.stringify(payload) 
      });
      toast(`预约创建成功，到期时间 ${expiryDate}`, 'success');
    }
    
    closeReservationFormModal();
    loadReservations();
  } catch (e) {
    // 处理重复预约错误
    if (e.message.includes('已经预约')) {
      toast('该读者已预约此书，请勿重复预约', 'error');
    } else if (e.message.includes('账号已冻结')) {
      toast('该读者账号状态异常，无法预约', 'error');
    } else {
      toast(e.message, 'error');
    }
  }
}
```

### 3. 完善导出CSV功能

更新 `exportReservations` 函数：

```javascript
async function exportReservations() {
  try {
    const status = $('reservationStatus').value || '';
    const keyword = $('reservationKeyword').value || '';
    const startDate = $('reservationStartDate')?.value || '';
    const endDate = $('reservationEndDate')?.value || '';
    
    let url = `/api/reservations/export?status=${encodeURIComponent(status)}&keyword=${encodeURIComponent(keyword)}`;
    if (startDate) url += `&start_date=${startDate}`;
    if (endDate) url += `&end_date=${endDate}`;
    
    // 显示加载提示
    const exportBtn = $('exportReservationsBtn');
    const originalText = exportBtn.textContent;
    exportBtn.textContent = '导出中...';
    exportBtn.disabled = true;
    
    const response = await fetch(`${API_BASE_URL}${url}`, {
      headers: { 'Authorization': `Bearer ${state.token}` }
    });
    
    if (!response.ok) throw new Error('导出失败');
    
    const blob = await response.blob();
    const downloadUrl = window.URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = downloadUrl;
    
    // 自动生成文件名：预约记录_YYYYMMDD.csv
    const today = new Date();
    const dateStr = today.getFullYear().toString() + 
                   (today.getMonth() + 1).toString().padStart(2, '0') + 
                   today.getDate().toString().padStart(2, '0');
    a.download = `预约记录_${dateStr}.csv`;
    
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    window.URL.revokeObjectURL(downloadUrl);
    
    toast('文件导出成功', 'success');
    
    // 恢复按钮状态
    exportBtn.textContent = originalText;
    exportBtn.disabled = false;
  } catch (e) {
    toast(e.message, 'error');
    // 恢复按钮状态
    const exportBtn = $('exportReservationsBtn');
    exportBtn.textContent = '导出 CSV';
    exportBtn.disabled = false;
  }
}
```
